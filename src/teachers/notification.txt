import { useEffect, useState } from "react";
import axios from "axios";
import { io } from "socket.io-client";
import {
  getChatsByTeacher,
  getTeacherMessages,
  uploadMessageFile,
} from "../api/chats"; // แก้ path ให้ตรงโปรเจค
import { getUserId, getFullName } from "../js/auth";
import Sidebar from "../components/sidebarAdmin";
import { Paperclip } from "lucide-react";

// ✅ socket สร้างครั้งเดียว (global)
const socket = io("http://localhost:3001", { withCredentials: true });

export default function CommunicationCenter() {
  const teacherId = Number(getUserId());
  const [students, setStudents] = useState([]);
  const [selectedStudent, setSelectedStudent] = useState(null);
  const [chatMessages, setChatMessages] = useState([]);
  const [input, setInput] = useState("");
  const [file, setFile] = useState(null);
  const [preview, setPreview] = useState(null);

  // โหลดห้องแชทของครู
  useEffect(() => {
    if (!teacherId) return;
    const fetchChats = async () => {
      const res = await getChatsByTeacher(teacherId);
      if (res.status === "success") setStudents(res.chats);
    };
    fetchChats();
  }, [teacherId]);

  // ฟังข้อความใหม่จาก socket
  useEffect(() => {
    socket.on("newMessage", (msg) => {
      // append เฉพาะห้องที่ครูกำลังเปิดอยู่
      if (msg.chat_id === selectedStudent?.chat_id) {
        setChatMessages((prev) => {
          if (prev.some((m) => m.message_id === msg.message_id)) return prev; // กันซ้ำ
          return [...prev, msg];
        });
      }
    });

    return () => {
      socket.off("newMessage");
    };
  }, [selectedStudent]);

  // เปิดห้องแชท
  const openChat = async (student) => {
    setSelectedStudent(student);

    // join room
    socket.emit("joinChat", student.chat_id);

    // โหลดข้อความเก่า
    const res = await getTeacherMessages(student.chat_id);
    if (res.status === "success") setChatMessages(res.messages);
  };

  // ส่งข้อความ
  const handleSend = async () => {
    if (!input.trim() && !file) return;
    if (!selectedStudent?.chat_id) return;

    let message_type = "text";
    let message_text = input;
    let file_url = null;

    if (file) {
      const uploadRes = await uploadMessageFile(file);
      if (uploadRes.status === "success") {
        message_type = uploadRes.file_type;
        message_text = "";
        file_url = uploadRes.file_url;
      } else {
        alert("Upload failed: " + uploadRes.message);
        return;
      }
    }

    const payload = new FormData();
    payload.append("chat_id", selectedStudent.chat_id);
    payload.append("sender_id", teacherId);
    payload.append("message_text", message_text);
    payload.append("message_type", message_type);
    payload.append("file_url", file_url || "");

    const res = await axios.post("/api/chats/sendMessage.php", payload, {
      withCredentials: true,
    });

    if (res.data.status === "success") {
      const newMsg = {
        message_id: res.data.message_id,
        chat_id: selectedStudent.chat_id,
        sender_id: teacherId,
        message_text,
        message_type,
        file_url,
        create_at: new Date().toISOString(),
        first_name: getFullName().split(" ")[0],
        last_name: getFullName().split(" ")[1] || "",
        role: 1, // ครู
      };

      // ✅ ส่งให้ทุกคนในห้อง (รวมตัวเอง)
      socket.emit("sendMessage", newMsg);
    }

    setInput("");
    setFile(null);
    setPreview(null);
  };

  // 🆕 ฟังก์ชันเลือกไฟล์แล้วแสดง preview
  const handleFileUpload = (file) => {
    if (!file) return;
    setFile(file);

    const previewUrl = URL.createObjectURL(file);
    setPreview({
      file,
      url: previewUrl,
      type: file.type.startsWith("image") ? "image" : "file",
    });
  };

  return (
    <div className="flex h-screen">
      <Sidebar />
      {/* Sidebar - รายชื่อนักเรียน */}
      <div className="w-1/4 border-r border-gray-200 p-4 overflow-y-auto">
        <h2 className="text-lg font-semibold mb-4">รายชื่อนักเรียน</h2>
        {students.map((student) => (
          <div
            key={student.chat_id}
            className={`p-2 cursor-pointer rounded ${selectedStudent?.chat_id === student.chat_id
              ? "bg-blue-100"
              : "hover:bg-gray-100"
              }`}
            onClick={() => openChat(student)}
          >
            {student.student_name}
          </div>
        ))}
      </div>

      {/* Chat window */}
      <div className="flex-1 flex flex-col">
        {selectedStudent ? (
          <>
            <div className="p-4 border-b border-gray-200 font-semibold">
              {selectedStudent.student_name}
            </div>

            {/* แสดงข้อความ */}
            <div className="flex-1 p-4 overflow-y-auto">
              {chatMessages.map((msg) => (
                <div
                  key={msg.message_id}
                  className={`mb-2 flex ${msg.sender_id === teacherId ? "justify-end" : "justify-start"
                    }`}
                >
                  {msg.message_type === "text" ? (
                    <div
                      className={`p-2 rounded-lg max-w-xs ${msg.sender_id === teacherId
                        ? "bg-blue-500 text-white"
                        : "bg-gray-200 text-black"
                        }`}
                    >
                      {msg.message_text}
                    </div>
                  ) : msg.message_type === "image" ? (
                    <img
                      src={`/api${msg.file_url}`}
                      alt="ส่งมา"
                      className="max-w-[200px] rounded"
                    />
                  ) : msg.message_type === "file" ? (
                    <a
                      href={`/api${msg.file_url}`}
                      target="_blank"
                      rel="noreferrer"
                      className="underline text-blue-600"
                    >
                      ดาวน์โหลดไฟล์
                    </a>
                  ) : null}
                </div>
              ))}
            </div>


            <div className="p-4 border-t border-gray-200">
              {/* preview file */}
              {preview && (
                <div className="mb-2">
                  {preview.type === "image" ? (
                    <img
                      src={preview.url}
                      alt="preview"
                      className="max-w-[150px] rounded"
                    />
                  ) : (
                    <div className="flex items-center space-x-2 p-2 border rounded bg-gray-100">
                      📎 <span>{preview.file.name}</span>
                    </div>
                  )}
                </div>
              )}

              {/* กล่องพิมพ์ */}
              <div className="flex items-center">
                {/* ปุ่มอัปโหลดไฟล์ */}
                <label
                  htmlFor="file-upload"
                  className="cursor-pointer mr-2 flex items-center text-gray-600 hover:text-blue-500"
                >
                  <Paperclip size={20} />
                </label>
                <input
                  id="file-upload"
                  type="file"
                  className="hidden"
                  onChange={(e) => handleFileUpload(e.target.files[0])}
                />

                {/* กล่องข้อความ */}
                <input
                  type="text"
                  value={input}
                  onChange={(e) => setInput(e.target.value)}
                  placeholder="พิมพ์ข้อความ..."
                  className="flex-1 border border-gray-200 rounded px-3 py-2 mr-2"
                />

                <button
                  onClick={handleSend}
                  className="bg-blue-500 text-white px-4 py-2 rounded"
                >
                  ส่ง
                </button>
              </div>
            </div>

          </>
        ) : (
          <div className="flex-1 flex items-center justify-center text-gray-500">
            เลือกห้องแชทจากด้านซ้าย
          </div>
        )}
      </div>
    </div>
  );
}
